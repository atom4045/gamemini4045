using UnityEngine; using System.Collections; [RequireComponent(typeof(Rigidbody2D))] public class PlayerController : MonoBehaviour { [Header("Movement")] public float moveSpeed = 10f; public float jumpForce = 14f; [Header("Combat")] public Transform attackPoint; public float attackRange = 0.8f; public LayerMask enemyLayers; public float attackCooldown = 0.3f; private Rigidbody2D rb; private bool isGrounded; private bool isAttacking; private float lastAttackTime; void Start() { rb = GetComponent(); } void Update() { // Ввод движения float moveInput = Input.GetAxisRaw("Horizontal"); if (!isAttacking) { rb.velocity = new Vector2(moveInput * moveSpeed, rb.velocity.y); // Поворот спрайта if (moveInput > 0) transform.localScale = new Vector3(1, 1, 1); else if (moveInput < 0) transform.localScale = new Vector3(-1, 1, 1); } // Прыжок if (Input.GetButtonDown("Jump") && isGrounded) { rb.velocity = new Vector2(rb.velocity.x, jumpForce); } // Атака if (Input.GetKeyDown(KeyCode.F) && Time.time >= lastAttackTime + attackCooldown) { StartCoroutine(PerformAttack()); } } IEnumerator PerformAttack() { isAttacking = true; lastAttackTime = Time.time; // Поиск врагов в радиусе Collider2D[] hitEnemies = Physics2D.OverlapCircleAll(attackPoint.position, attackRange, enemyLayers); foreach (Collider2D enemy in hitEnemies) { Debug.Log("Удар по: " + enemy.name); // Добавь здесь: enemy.GetComponent().TakeDamage(10); } yield return new WaitForSeconds(attackCooldown); isAttacking = false; } // Проверка земли (нужен объект с тегом "Ground") void OnCollisionEnter2D(Collision2D col) { if (col.gameObject.CompareTag("Ground")) isGrounded = true; } void OnCollisionExit2D(Collision2D col) { if (col.gameObject.CompareTag("Ground")) isGrounded = false; } // Визуализация радиуса атаки в редакторе void OnDrawGizmosSelected() { if (attackPoint == null) return; Gizmos.DrawWireSphere(attackPoint.position, attackRange); } }

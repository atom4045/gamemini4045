using UnityEngine; [RequireComponent(typeof(Rigidbody2D), typeof(AudioSource))] public class GameManager : MonoBehaviour { [Header("Settings")] public float moveSpeed = 6f; [Header("Combat")] public GameObject bulletPrefab; public Transform shootPoint; [Header("VFX & Audio")] public ParticleSystem shootEffect; public AudioSource shootAudio; private Rigidbody2D rb; void Start() { rb = GetComponent(); } void Update() { // 1. Движение float moveX = Input.GetAxis("Horizontal"); float moveY = Input.GetAxis("Vertical"); rb.velocity = new Vector2(moveX, moveY).normalized * moveSpeed; // 2. Поворот персонажа к мыши Vector3 mousePos = Camera.main.ScreenToWorldPoint(Input.mousePosition); Vector2 lookDir = (Vector2)mousePos - rb.position; float angle = Mathf.Atan2(lookDir.y, lookDir.x) * Mathf.Rad2Deg; rb.rotation = angle; // 3. Стрельба if (Input.GetMouseButtonDown(0)) { Fire(); } } void Fire() { // Создаем пулю Instantiate(bulletPrefab, shootPoint.position, shootPoint.rotation); // Запускаем эффекты if (shootEffect != null) { shootEffect.Play(); } // Запускаем звук if (shootAudio != null) { shootAudio.Play(); } } }
